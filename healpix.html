  • it told me how many bits are needed to store the Npix (even though it's easy to figure out)
  • the area is expressed in degrees, since that's how all of MY astronomer friends refer to it
  • the numbers were prettier-printed

For Future Melissa, here's the code I used to create this kind of table: 

import numpy as np
from tabulate import tabulate

import astropy.units as u

def order2npix(order: int) -> int:
    return 12 * (1 << (2 * order))
def order2resol(order: int, *, arcmin: bool = False, unit=u.rad) -> float:
    if arcmin:
        unit = u.arcmin
    unit = u.Unit(unit)

    return np.sqrt(order2pixarea(order, unit=unit * unit))

def order2pixarea(order: int, *, degrees: bool = False, unit=u.sr) -> float:
        if degrees:
            unit = "deg**2"
        unit = u.Unit(unit)

        npix = order2npix(order)
        pix_area_rad = 4 * np.pi / npix * u.steradian
        return pix_area_rad.to_value(unit)

def pretty_print_area(deg):
    if deg >= 1:
        return f"{deg:.2f} deg sq"
    elif (deg * 3600) >= 1:
        return f"{(deg * 3600):.2f} arcmin sq"
    elif (deg * 12960000) >= .01:
        return f"{(deg * 12960000):.2f} arcsec sq"
    else:        
        return f"{(deg * 12960000)} sec"

def pretty_print_angle(deg):
    if deg >= 1:
        return f"{deg:.2f} deg"
    elif (deg * 60) >= 1:
        return f"{(deg * 60):.2f} arcmin"
    elif (deg * 3600) >= .01:
        return f"{(deg * 3600):.2f} arcsec"
    else:        
        return f"{(deg * 3600)} arcsec"
    
orders = np.arange(21)
nsides = [int(2**i) for i in orders]
npixen = [order2npix(i) for i in orders]
bitses = [int(np.ceil(np.log2(npix))) for npix in npixen]
areaug = [order2pixarea(i, degrees=True) for i in orders]
areapp = [pretty_print_area(area) for area in areaug]
avgwdh = [order2resol(i, unit=u.deg) for i in orders]
wdthpp = [pretty_print_angle(i) for i in avgwdh]

print(tabulate({"Norder": orders,
                "Nsides": nsides,
                "Npix": npixen,
                "Bits needed": bitses,
                "Area": areapp,
                "Average width": wdthpp,
               }, headers="keys",intfmt=",",tablefmt="html",
               colalign=("right","right","right","right","right", "right")))
